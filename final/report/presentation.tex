\begin{frame}{Introduction \& Project Scope}
    \begin{itemize}
        \item \textbf{What is a Spanner?} A sparse subgraph $H \subseteq G$ such that distances are approximated within a factor $t \geq 1$.
        \item \textbf{Mathematical Definition:} $\forall u, v \in V: d_H(u, v) \leq t \cdot d_G(u, v)$

% 3. HISTORY OF SPANNERS
% -------------------------
\begin{frame}{The History \& Evolution of Graph Spanners}
    \begin{itemize}
        \item \textbf{1985-1988 (The Implicit Era):} Awerbuch introduced Network Synchronizers, partitioning graphs into bounded diameter clusters.
        \item \textbf{1989 (Formalization):} Peleg and Schäffer formally defined the concept of the "Graph Spanner." They exposed the deep relationship between sparsity and mathematical girth.
        \item \textbf{1993 (The Greedy Era):} Althöfer introduced the \textbf{Greedy BFS Spanner}. It produced the theoretically sparsest spanner $\mathcal{O}(n^{1+1/k})$ edges, but ran in cripplingly slow time.
        \item \textbf{2007 (The Breakthrough):} Surender Baswana and Sandeep Sen discovered a randomized, linear-time $\mathcal{O}(km)$ algorithm.
    \end{itemize}
\end{frame}

% -------------------------
% 4. THEORETICAL FOUNDATIONS
% -------------------------
\begin{frame}{Theoretical Foundations}
    \begin{itemize}
        \item \textbf{The Erd\H{o}s Girth Conjecture: The Optimality Limit}
        \item Spanners rely entirely on graph cycles. If we remove an edge $(u,v)$, the shortest alternate path must be $\leq t$.
        \item A graph has girth $g$ if its shortest cycle has length $g$. By definition, any girth $2k+1$ graph is a $(2k-1)$-spanner of itself.
        \item The Moore Bound tells us the theoretical minimum number of edges required to maintain connectivity without introducing excessive stretch. 
        \item Baswana-Sen effectively matches this hardware limit while remaining $\mathcal{O}(km)$ computationally.
    \end{itemize}
\end{frame}

% -------------------------
% 3.5 THE GREEDY SPANNER 
% -------------------------
\begin{frame}{Algorithm 1: Greedy Spanner - Core Idea}
    \begin{itemize}
        \item Sort edges by increasing weight, then add only the edges that are necessary to preserve stretch.
        \item For each edge $(u,v)$, compare the current spanner distance $d_H(u,v)$ with the allowed threshold $t \cdot w(u,v)$.
        \item If $d_H(u,v) > t \cdot w(u,v)$, the edge is inserted into the spanner.
        \item This produces a very sparse spanner, but the repeated shortest-path checks make it expensive.
        \item Mathematical form: $E' = \{(u,v) \in E \mid d_H(u,v) > t \cdot w(u,v)\}$.
    \end{itemize}
\end{frame}

% 5. BASWANA SEN ALGORITHM LOGIC
% -------------------------
\begin{frame}{Algorithm 2: Baswana-Sen - Core Idea}
    \begin{itemize}
        \item Randomly sample cluster centers in phases, rather than checking every edge globally.
        \item Each phase shrinks the graph by linking vertices to sampled centers, so the spanner stays sparse.
        \item The algorithm achieves expected size $O(k n^{1+1/k})$ and stretch $(2k-1)$.
        \item The sampling probability is typically $n^{-1/k}$ for center selection.
        \item This is the key reason the algorithm runs in linear time in $m$.
    \end{itemize}
\end{frame}

% 6. HAS ALGORITHM
% -------------------------
\begin{frame}{Algorithm 3: Hybrid Adaptive Spanner - Core Idea}
    \begin{itemize}
        \item HAS starts from Baswana-Sen but changes how centers are chosen.
        \item Vertices with higher degree get higher sampling priority, so hubs are favored in dense graphs.
        \item After skeleton construction, a local greedy pruning step removes redundant cycle edges.
        \item One useful summary is $P(v) = \min\!\left(1, \frac{deg(v)}{deg_{max}} \cdot n^{-1/k}\right)$.
        \item This makes HAS more topology-aware than the baseline randomized sampler.
    \end{itemize}
\end{frame}

\begin{frame}{The Literature Landscape}
    \textbf{Mapping our work to the academic ecosystem:}
    \vspace{0.4cm}
    \begin{itemize}
        \item \textbf{Multiplicative Spanners (This Project):} Strict $t$ multipliers (Baswana-Sen, Peleg).
        \item \textbf{Additive Spanners:} Distances stretched by a flat constant ($+2, +6$) instead of a multiplier. Required for extremely dense core networks.
        \item \textbf{Fault-Tolerant Spanners:} Redundant substructures specifically modeled to guarantee routing if $f$ nodes are simultaneously destroyed.
    \end{itemize}
\end{frame}

% -------------------------
% 16. THEORY VS PRACTICE
% -------------------------
\begin{frame}{The Theory vs. Practice Gap}
    \begin{itemize}
        \item \textbf{The Stretch Paradox:} Theoretical papers promise bound $(2k-1)$, leading developers to expect $stretch=5$ for $k=3$. In reality, average empirical stretch rarely passes $1.3$. Real-world graphs naturally suppress algorithmic damage.
        \item \textbf{The Bottleneck of Planarity:} Research assumes average random distributions. Spatial layouts like physical Road Networks violate these clustering expectations, causing theoretical sparsity guarantees to drastically underperform in reality.
    \end{itemize}
\end{frame}
